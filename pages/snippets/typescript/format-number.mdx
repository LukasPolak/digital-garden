# Format Number

Utility function to format number

```ts
export const formatNumber = (number: number) => {
  if (!number) {
    return "";
  }

  try {
    return new Intl.NumberFormat("en-US", {
      notation: "compact",
    }).format(number);
  } catch (e) {
    console.error(e);
    return "";
  }
};
```

## Usage

```ts
formatNumber(1000); // 1K
```
